const not = (n) => !n;

/**
 * @param {number} num
 * @return {boolean}
 */
var checkPerfectNumber = function(num) {
    const divisors = [];
    for(let i = 0; i <= num / 2; i++){
        if(not(num % i)) divisors.push(i);
    }
    return divisors.reduce((a,c)=>a+c, 0) === num;
};
