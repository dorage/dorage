/**
 * @param {number[]} prices
 * @return {number}

var maxProfit = function(prices) {
    const len = prices.length;
    let result = 0;
    let i = 0;
    
    while(i < len) {
        while(i < len && prices[i + 1] <= prices[i]) i++;
        const buy = prices[i];
        
        while(i < len && prices[i + 1] > prices[i]) i++;
        const sell = prices[i];
        
        result += sell - buy;
        i++;
    }
    
    return result;
};
