/**
 * @param {number[]} target
 * @param {number[]} arr
 * @return {boolean}
 */
var canBeEqual = function(target, arr) {
    const memo = target.reduce((acc, curr)=>{
        if(!acc[curr]) acc[curr] = 0;
        acc[curr]++;
        return acc;
    },{})
    
    for(let i = 0, len = arr.length; i < len; i++){
        const curr = arr[i];
        if(!memo[curr]) return false;
        memo[curr]--;
    }
    return true;
};
