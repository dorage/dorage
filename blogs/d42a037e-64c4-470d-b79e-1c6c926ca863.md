function getMassOfMolecule(fomula) {
    const stack = [];

    for (let i = 0; i < fomula.length; i++) {
        const atom = fomula[i];
        // ìˆ«ìì¼ ë•Œ
        if (!isNaN(Number(atom))) {
            stack.push(atom * stack.pop());
            continue;
        }
        // ë¬¸ìì¼ ë•Œ
        switch (atom) {
            case 'H':
                stack.push(1);
                continue;
            case 'C':
                stack.push(12);
                continue;
            case 'O':
                stack.push(16);
                continue;
        }
        // ê´„í˜¸ì¼ë•Œ
        i++;
        let bracket = 1;
        let partial = '';
        for (i; i < fomula.length; i++) {
            const atom = fomula[i];
            if (atom === '(') {
                bracket++;
            }
            if (atom === ')') {
                bracket--;
                if (bracket === 0) {
                    stack.push(getMassOfMolecule(partial));
                    break;
                }
            }
            partial += atom;
        }
    }

    return stack.reduce((a, c) => a + c, 0);
}
