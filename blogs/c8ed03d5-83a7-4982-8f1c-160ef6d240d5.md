 * @param {string} s
 * @param {number} k
 * @return {string}

var orderlyQueue = function(s, k) {
    if(k > 1) {
        return [...s].sort().join('');
    }
    
    let ss = s;
    for(let i = 1; i < ss.length; i++) {
        const temp = s.slice(i) + s.slice(0, i)
        ss = ss <= temp ? ss : temp;
    }
    return ss;
};
