/**
 * @param {number[]} arr
 * @return {number}
 */
var peakIndexInMountainArray = function(arr) {
    let max = 0;
    for(let i = 0; i < arr.length; i++){
        const e = arr[i];
        if(e < max) return i - 1;
        max = e;
    }
};

/**
 * @param {number[]} arr
 * @return {number}
 */
var peakIndexInMountainArray = function(arr) {
    let left = 0, right = arr.length - 1;
    while(left < right){
        const mid = Math.floor((left+right)/2);
        if(arr[mid] < arr[mid + 1]){
            left = mid + 1;
        } else {
            right = mid;
        }
    }
    return left;
};
