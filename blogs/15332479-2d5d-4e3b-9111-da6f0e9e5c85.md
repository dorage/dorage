```python
class Solution:
    def getWordsInLongestSubsequence(self, n: int, words: List[str], groups: List[int]) -> List[str]:
        def is_ham(s1:str, s2:str):
            if len(s1) != len(s2):
                return False

            diff = 0
            for i in range(len(s1)):
                if not s1[i] == s2[i]:
                    diff += 1
                if diff > 1:
                    return False

            return True

        def is_adjacent(i: int, j: int):
            if j < 0:
                return True
            if groups[i] == groups[j]:
                return False
            return is_ham(words[i], words[j])
                
        @cache
        def recursive(i: int, j: int):
            if i >= n:
                return []
            
            r1 = recursive(i + 1, j)
            if is_adjacent(i, j):
                r2 = [words[i], *recursive(i + 1, i)]
                if len(r1) < len(r2):
                    return r2
            return r1
        
        return recursive(0, -1)
```

```python
class Solution:
    def getWordsInLongestSubsequence(self, n: int, words: List[str], groups: List[int]) -> List[str]:
        def is_adjacent(i:int, j:int):
            if groups[i] == groups[j]:
                return False
            if len(words[i]) != len(words[j]):
                return False
            if sum([c1 != c2 for c1, c2 in zip(words[i], words[j])]) != 1:
                return False
            return True

        @cache
        def dp(i: int):
            pik = []

            for j in range(i + 1, n):
                if is_adjacent(i, j):
                    pik = max(pik, dp(j), key=len)

            return [words[i]] + pik
                    
            

        return max([dp(i) for i in range(n)], key=len)
```
