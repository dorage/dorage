/**

var letterCasePermutation = function(s) {
    let answer = [''];
    for(const char of [...s]) {
        if(isNaN(char)) {
            answer = [...answer.map(e => e + char.toLowerCase()), ...answer.map(e => e + char.toUpperCase())];
        } else {
            answer = answer.map(e => e + char);
        }
    }
    return answer;
};
