/**
 * @param {number} n
 * @param {number[][]} edges
 * @return {number[]}
 */
var findSmallestSetOfVertices = function(n, edges) {
    const end = Array(n).fill(false);
    const answer = [];
    
    for(const [a,b] of edges) {
        end[b] = true;
    }
    
    for(let i = 0; i < n; i++) {
        if(end[i]) continue;
        answer.push(i);
    }
    
    return answer;
};
