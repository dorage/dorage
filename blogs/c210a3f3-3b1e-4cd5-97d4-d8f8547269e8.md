const readline = require('readline');

let input = [];

readline
    .createInterface({
        input: process.stdin,
        output: process.stdout,
    })
    .on('line', function (line) {
        input.push(line);
    })
    .on('close', function () {
        solution(input);
        process.exit();
    });

function solution(input) {
    const n = Number(input.shift());
    const x = log3(n);
    console.log(
        drawStar(x)
            .map((e) => e.map((e) => (e === 1 ? '*' : ' ')).join(''))
            .join('\n'),
    );
}

function log3(n) {
    let x = 0;
    while (n > 1) {
        n = n / 3;
        x++;
    }
    return x;
}

function drawBlank(n) {
    if (n <= 1)
        return [
            [0, 0, 0],
            [0, 0, 0],
            [0, 0, 0],
        ];

    const map = [];
    for (let i = 0; i < 3; i++) {
        const row = [];
        for (let j = 0; j < 3; j++) {
            const elem = drawBlank(n - 1);
            if (!row.length) {
                row.push(...elem);
                continue;
            }
            row.forEach((arr, idx) => {
                arr.push(...elem[idx]);
            });
        }
        map.push(...row);
    }
    return map;
}

function drawStar(n) {
    if (n <= 1)
        return [
            [1, 1, 1],
            [1, 0, 1],
            [1, 1, 1],
        ];

    const map = [];
    for (let i = 0; i < 3; i++) {
        const row = [];
        for (let j = 0; j < 3; j++) {
            const elem =
                i === 1 && j === 1 ? drawBlank(n - 1) : drawStar(n - 1);
            if (!row.length) {
                row.push(...elem);
                continue;
            }
            row.forEach((arr, idx) => {
                arr.push(...elem[idx]);
            });
        }
        map.push(...row);
    }
    return map;
}
