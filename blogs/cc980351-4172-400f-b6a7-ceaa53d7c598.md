const getMN = (grid) => [grid.length, grid[0].length];

/**
 * @param {number[][]} grid
 * @return {number}
 */
var numEnclaves = function(grid) {
    const [m, n] = getMN(grid);
    let answer = 0;
    
    for(let i = 0; i < m; i++) {
        for(let j = 0; j < n; j++) {
            if(!grid[i][j]) continue;
            answer += BFS(grid, i, j);
        }
    }
    
    return answer;
};

const BFS = (grid, i, j) => {
    let island = true;
    let size = 0;
    const [m, n] = getMN(grid);
    
    const q = [[i, j]];
    
    while(q.length) {
        const preq = [];
        while(q.length) {
            const [r, c] = q.shift();
            
            if(!grid[r][c]) continue;
            grid[r][c] = 0;
            
            size++;
            if(r === 0) island = false;
            if(r === m - 1) island = false;
            if(c === 0) island = false;
            if(c === n - 1) island = false;
            
            if(r > 0 && grid[r - 1][c]) preq.push([r - 1, c]);
            if(r + 1 < m && grid[r + 1][c]) preq.push([r + 1, c]);
            if(c > 0 && grid[r][c - 1]) preq.push([r, c - 1]);
            if(c + 1 < n && grid[r][c + 1]) preq.push([r, c + 1]);
        }
        q.push(...preq);
    }
    
    return island ? size : 0;
}
