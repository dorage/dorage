function matrixChainMultiplication(n, A, result = 0) {
    if (A.length <= 1) {
        return result;
    }
    const results = [];
    for (let i = 0; i < n - 1; i++) {
        const copy = [...A];
        const [x1, y1] = copy[i];
        const [_, y2] = copy[i + 1];
        const newA = [...copy.slice(0, i), [x1, y2], ...copy.slice(i + 2, n)];
        results.push(
            matrixChainMultiplication(n - 1, newA, x1 * y1 * y2 + result),
        );
    }
    return Math.min(...results);
}
