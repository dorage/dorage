 * Definition for singly-linked list.
 * function ListNode(val, next) {

/**
 * @param {ListNode} head
 * @return {ListNode}
 */
var deleteMiddle = function(head) {
    if(!head.next) return null;
    if(!head.next.next) {
        head.next = null;
        return head;
    }
    
    let slow = head;
    let fast = head.next.next;
    
    while(fast.next) {
        fast = fast.next;
        slow = slow.next;
        if(!fast.next) break;
        fast = fast.next;
    }
    
    slow.next = slow.next.next;
    return head;
};
