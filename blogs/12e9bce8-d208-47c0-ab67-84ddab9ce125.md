const getMN = (mat) => [mat.length, mat[0].length];

/**
 * @param {number[][]} mat
 * @param {number} k
 * @return {number[][]}
 */
var matrixBlockSum = function(mat, k) {
    const [m, n] = getMN(mat);
    const answer = Array(m).fill().map(() => Array(n).fill(0));
    
    // [0, 0]
    for(let i = 0; i <= k; i++) {
        for(let j = 0; j <= k; j++) {
            if(i >= m || j >= n) continue;
            answer[0][0] += mat[i][j];
        }
    }
    
    let sum = answer[0][0];
    for(let i = 0; i < m; i++) {
        for(let j = 1; j < n; j++) {
            const dx = k + 1;
            // ì—´ ê³ ì •, ì–‘ ë ì—´ë§Œ í•©ì‚°
            for(let dy = -k; dy <= k; dy++) {
                const mx = j - dx; // minus
                const px = j + k; // plus
                const y = i + dy;
                
                if(y < 0 || y >= m) continue;
                
                if(mx >= 0) sum -= mat[y][mx];
                if(px < n) sum += mat[y][px];
            }
            
            answer[i][j] = sum;
        }
        
        if(i === m - 1) continue;
        
        sum = answer[i][0];
        // í–‰ ê³ ì •, ì–‘ ë í–‰ë§Œ í•©ì‚°
        const dy = k;
        for(let dx = 0; dx <= k; dx++) {
            const my = i - dy;
            const py = i + k + 1;
            const x = dx;
            
            if(x < 0 || x >= n) continue;

            if(my >= 0) sum -= mat[my][x];
            if(py < m) sum += mat[py][x];
        }
        answer[i + 1][0] = sum;
    }
    
    return answer;
};
