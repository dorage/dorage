const readline = require('readline');

let input = [];

readline
    .createInterface({
        input: process.stdin,
        output: process.stdout,
    })
    .on('line', function (line) {
        input.push(line);
    })
    .on('close', function () {
        solution(input);
        process.exit();
    });

function solution(input) {
    const m = Number(input.shift());
    const tree = Tree();
    for (const line of input) {
        if (line === 'print') {
            tree.print();
            continue;
        }
        const [_, id] = line.split(' ');
        tree.insert(Number(id));
    }
}

function Tree() {
    if (!new.target) return new Tree();
    this.root = null;
}
Tree.prototype.insert = function (id) {
    if (!this.root) {
        this.root = Node(id);
        this.root.parent = -1;
        return;
    }
    let curr = this.root;
    while (true) {
        if (id < curr.id) {
            if (!curr.left) {
                curr.left = Node(id);
                curr.left.parent = curr;
                break;
            }
            curr = curr.left;
        } else {
            if (!curr.right) {
                curr.right = Node(id);
                curr.right.parent = curr;
                break;
            }
            curr = curr.right;
        }
    }
};
Tree.prototype.print = function () {
    console.log(
        ' ' +
            this.inorder()
                .map((e) => e.id)
                .join(' '),
    );
    console.log(
        ' ' +
            this.preorder()
                .map((e) => e.id)
                .join(' '),
    );
};

Tree.prototype.preorder = function () {
    const queue = [this.root];
    const preorder = [];
    while (queue.length) {
        const curr = queue.shift();
        preorder.push(curr);
        if (curr.right) queue.unshift(curr.right);
        if (curr.left) queue.unshift(curr.left);
    }
    return preorder;
};

Tree.prototype.inorder = function () {
    const queue = [];
    let curr = this.root;
    const inorder = [];

    while (curr || queue.length) {
        while (curr) {
            queue.push(curr);
            curr = curr.left;
        }
        curr = queue.pop();
        inorder.push(curr);
        curr = curr.right;
    }
    return inorder;
};

function Node(id) {
    if (!new.target) return new Node(id);
    this.id = id;
    this.left = null;
    this.right = null;
    this.parent = null;
}
