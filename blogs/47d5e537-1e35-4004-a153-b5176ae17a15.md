function solution(input) {
    const memo = [1, 1];
    const n = Number(input.shift());
    return Fibonacci(memo, n);
}

function Fibonacci(memo, n) {
    if (memo[n]) return memo[n];

    const curr = (2 * Fibonacci(memo, n - 2) + Fibonacci(memo, n - 1)) % 10007;
    memo[n] = curr;
    return curr;
}
