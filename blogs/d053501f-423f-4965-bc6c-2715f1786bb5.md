var findArray = function(pref) {
    const arr = [pref[0]];
    let xorAcc = pref[0];
    for(let i = 1; i < pref.length; i++) {
        arr.push(xorAcc ^ pref[i]);
        xorAcc = xorAcc ^ arr[i];
    }
    return arr;
};
