function swap(A, i, j) {
    const temp = A[i];
    A[i] = A[j];
    A[j] = temp;
}

function dutchNationalFlag(A) {
    let start = 0;
    let mid = 0;
    let end = A.length - 1;
    const pivot = 1;
    while (mid <= end) {
        if (A[mid] > pivot) {
            // 2
            swap(A, mid, end);
            end--;
        } else if (A[mid] === pivot) {
            // 1
            mid++;
        } else {
            // 0
            swap(A, start, mid);
            start++;
            mid++;
        }
    }
}
