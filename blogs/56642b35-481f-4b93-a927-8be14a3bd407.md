function topKFrequent(words: string[], k: number): string[] {
    const map = {};
    for(let i = 0; i < words.length; i++){
        const word = words[i];
        if(!map[word]) map[word] = 0;
        map[word]++;
    }
    
    const keys = Object.keys(map);
    keys.sort();
    keys.sort((a, b) => {
        return map[b] - map[a];
    })
    
    return keys.slice(0, k);
};
