/**

var permute = function(nums) {
    return recursive(nums);
};

const recursive = (nums) => {
    if(!nums.length) return [[]];
    const arr = [];
    for(let i = 0; i < nums.length; i++) {
        const curr = [...nums];
        curr.splice(i, 1);
        arr.push(...recursive(curr).map(e=>[nums[i], ...e]));
    }
    return arr;
};
