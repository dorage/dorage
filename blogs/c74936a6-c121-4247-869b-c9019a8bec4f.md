/**
 * @param {number[]} tokens
 * @param {number} power
 * @return {number}
 */
var bagOfTokensScore = function(tokens, power) {
    tokens.sort((a,b) => a- b);
    
    let answer = 0;
    let score = 0;
    let l = 0;
    let r = tokens.length - 1;
    
    while(l <= r){
        if(tokens[l] > power) {
            if(!score) break;
            power += tokens[r];
            r--;
            score--;
        } else {
            power -= tokens[l];
            l++;
            score++;
        }
        answer = Math.max(answer, score);
    }
    
    return answer;
};
