var combine = function(n, k) {
    const answer = [];
    for(let i = 1; i <= n - k + 1; i++) {
        answer.push(...combination(i, n, k - 1));
    }
    return answer;
};

const combination = (s, n, k) => {
    if(!k) return [[s]];
    const arr = [];
    for(let i = s + 1; i <= n - k + 1; i++){
        arr.push(...combination(i, n, k - 1).map((e)=>[s, ...e]));
    }
    return arr;
};
