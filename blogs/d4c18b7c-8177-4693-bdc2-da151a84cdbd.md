const GEN = ['A', 'C', 'G', 'T']

/**
 * @param {string} start
 * @param {string} end
 * @param {string[]} bank
 * @return {number}
 */
var minMutation = function(start, end, bank) {
    const set = new Set(bank);
    if(!set.has(end)) return -1;
    
    const q = [start];
    const visited = new Set();
    let count = 0;
    while(q.length) {
        const preq = [];
        while(q.length) {
            const str = q.shift();
            
            if(str === end) return count;
            if(visited.has(str)) continue;
            visited.add(str);
            
            preq.push(...getPossibleMutations(str, set));
        }
        count++;
        q.push(...preq);
    }
    
    return -1;
};

const getPossibleMutations = (str, possibles, visited) => {
    const muts = [];
    for(let i = 0; i < 8; i++) {
        for(const c of GEN) {
            const mut = str.slice(0, i) + c + str.slice(i + 1, 8);
            if(mut === str) continue;
            if(possibles.has(mut)) muts.push(mut);
        }
    }
    return muts;
};
