function deleteAndEarn(nums: number[]): number {
    const map = nums.reduce((acc, curr) => {
        if(!acc[curr]) acc[curr] = 0;
        acc[curr] += curr;
        return acc;
    }, {});
    
    let take = 0;
    let skip = 0;
    for(let i = 0; i <= 10000; i++){
        const takei = skip + ( map[i] ? map[i] : 0 );
        const skipi = Math.max(skip, take);
        take = takei;
        skip = skipi;
    }
    return Math.max(take, skip);
};
