/**
 * @param {string} s
 * @param {string[]} wordDict
 * @return {boolean}
 */
var wordBreak = function(s, wordDict) {
    const set = new Set(wordDict);
    const memo = {};
    return wb(s, set, memo);
};

const wb = (s, set, memo) => {
    if(memo[s] == null) {
        const len = s.length;
        if(!len) {
            memo[s] = true;
            return memo[s];
        }
        for(let i = 1; i <= len; ++i) {
            if(set.has(s.slice(0, i)) && wb(s.slice(i), set, memo)) {
                memo[s] = true;
                return memo[s];
            }
        }
        memo[s] = false;
        return memo[s];
    }
    return memo[s];
}
