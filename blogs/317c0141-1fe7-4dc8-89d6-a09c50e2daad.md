function solution(input) {
    const n = Number(input.shift());
    const commands = input.map((el) =>
        el.split(' ').map((el) => (isNaN(Number(el)) ? el : Number(el))),
    );
    linkedList(n, commands);
}

function linkedList(n, commands) {
    const list = new doublyLinkedList();

    for (const command of commands) {
        const [exec, m] = command;
        switch (exec) {
            case 'insert':
                list.insert(m);
                break;
            case 'delete':
                list.delete(m);
                break;
            case 'deleteFirst':
                list.deleteFirst();
                break;
            case 'deleteLast':
                list.deleteLast();
                break;
        }
    }

    list.print();
}

function Node(n) {
    if (!new.target) return new Node(n);
    this.n = n;
    this.prev = null;
    this.next = null;
}

function doublyLinkedList(n, commands) {
    if (!new.target) return new doublyLinkedList(n);
    this.first;
    this.last;
}

doublyLinkedList.prototype.insert = function (n) {
    const node = new Node(n);
    if (!this.first) {
        this.first = node;
        this.last = node;
        return;
    }
    node.next = this.first;
    this.first.prev = node;
    this.first = node;
};

doublyLinkedList.prototype.delete = function (n) {
    if (!this.first) return;
    let curr = this.first;

    while (curr) {
        if (curr.n === n) {
            if (curr.next && curr.prev) {
                curr.next.prev = curr.prev;
                curr.prev.next = curr.next;
            }
            if (!curr.prev) {
                this.deleteFirst();
            }
            if (!curr.next) {
                this.deleteLast();
            }
            return;
        }
        curr = curr.next;
    }
};

doublyLinkedList.prototype.deleteFirst = function () {
    this.first.next.prev = null;
    this.first = this.first.next;
};

doublyLinkedList.prototype.deleteLast = function () {
    this.last.prev.next = null;
    this.last = this.last.prev;
};

doublyLinkedList.prototype.print = function () {
    const answer = [];
    let curr = this.first;
    while (curr) {
        answer.push(curr.n);
        curr = curr.next;
    }
    console.log(answer.join(' '));
};
