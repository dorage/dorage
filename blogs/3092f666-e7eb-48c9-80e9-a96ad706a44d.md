/**

 * @param {number} maximumBit

var getMaximumXor = function(nums, maximumBit) {
    const max = Math.pow(2, maximumBit) - 1;
    const answer = [];
    
    const last = nums.reduce((a,c)=>{
        answer.unshift(a ^ max);
        return a ^ c;
    }, nums.shift());
    answer.unshift(last ^ max);
    
    return answer;
};
