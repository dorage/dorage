/**
 * Definition for singly-linked list.
 * function ListNode(val) {
 *     this.val = val;
 *     this.next = null;
 * }
 */

/**
 * @param {ListNode} head
 * @return {ListNode}
 */
var detectCycle = function(head) {
    let slow = head;
    let fast = head;
    let isCycle = false;
    
    while(fast && fast.next && fast.next.next){
        slow = slow.next;
        fast = fast.next.next;
        
        if(slow === fast){
            while(head !== slow){
                head = head.next;
                slow = slow.next;
            }
            return head;
        }
    }
    return null;
};
