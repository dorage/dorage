/**

var maxProduct = function(nums) {
    let answer = nums[0];
    let min = answer;
    let max = answer;
    
    for(let i = 1; i < nums.length; i++) {
        const n = nums[i];
        
        if(n < 0) [max, min] = [min, max];
        
        max = Math.max(n, n * max);
        min = Math.min(n, n * min);
        
        answer = Math.max(answer, max);
    }
    
    return answer;
};
