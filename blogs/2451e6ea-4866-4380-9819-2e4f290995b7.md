var concatenatedBinary = function(n) {
    const MOD = 1000000007n;
    
    let sum = 0n;
    let length = 0n;
    
    for(let i = 1n; i <= n; i++){
        if((i & (i - 1n)) === 0n) length++;
        sum = ((sum << length) | i) % MOD;
    }
    
    return sum;
};
