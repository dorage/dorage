/**
 * @param {number[]} nums
 * @param {number} k
 * @return {boolean}
 */
var checkSubarraySum = function(nums, k) {
    if(nums.length < 2) return false;
    
    const map = {};
    let sum = 0;
    
    for(let i = 0; i < nums.length; i++) {
        sum += nums[i];
        const remainder = sum % k;
        
        if(i > 0 && !remainder) return true;
        
        if(!map[remainder]) {
            map[remainder] = i + 1;
        } else if (map[remainder] < i) {
            return true;
        }
    }
    
    return false;
};
