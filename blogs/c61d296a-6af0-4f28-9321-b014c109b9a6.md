/**
 * @param {number[]} nums
 * @return {boolean}
 */
var increasingTriplet = function(nums) {
    let min = Infinity, max = Infinity;
    
    for(const n of nums) {
        if(n <= min) {
            min = n;
        } else if (n <= max) {
            max = n;
        } else {
            return true;
        }
    }
    
    return false;
};
