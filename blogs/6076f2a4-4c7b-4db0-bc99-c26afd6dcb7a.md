/**
 * @param {number[]} arr
 * @param {number} start
 * @return {boolean}
 */
var canReach = function(arr, start) {
    const len = arr.length;
    const visited = Array(len).fill(false);
    
    const q = [start];
    
    while(q.length) {
        const preq = [];
        while(q.length) {
            const i = q.shift();
            const n = arr[i];
            
            if(n === 0) return true;
            
            if(visited[i]) continue;
            visited[i] = true;
            
            if(i + n < len && !visited[i + n]) preq.push(i + n);
            if(i - n >= 0 && !visited[i - n]) preq.push(i - n);
        }
        q.push(...preq);
    }
    
    return false;
};
