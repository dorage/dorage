function Tree() {
    if (!new.target) return new Tree();
    this.root = null;

Tree.prototype.insert = function (id) {
    if (!this.root) {
        this.root = Node(id);
        this.root.parent = -1;
        return;
    }
    let curr = this.root;
    while (true) {
        if (id < curr.id) {
            if (!curr.left) {
                curr.left = Node(id);
                curr.left.parent = curr;
                break;
            }
            curr = curr.left;
        } else {
            if (!curr.right) {
                curr.right = Node(id);
                curr.right.parent = curr;
                break;
            }
            curr = curr.right;
        }
    }
};

Tree.prototype.find = function (k) {
    let curr = this.root;
    while (curr.id !== k) {
        if (k < curr.id) {
            curr = curr.left;
        } else {
            curr = curr.right;
        }
        if (!curr) return null;
    }
    return curr;
};

Tree.prototype.delete = function (k) {
    const getMinimum = (node)=>{
        let target = node;
        while(target.left){
            target = target.left;
        }
        return target;
    };
    const getMaximum = (node)=>{
        let target = node;
        while(target.right){
            target = target.right;
        }
        return target;
    };
    const curr = this.find(k);

            this.root = null;

            curr.parent.left = null
            curr.parent = null;
        } else {
            curr.parent.right = null
            curr.parent = null;
        }
        return;
    }
    
    let target;
    if(curr.left === null){
        target = getMinimum(curr.right);
    } else if (curr.right === null){
        target = getMaximum(curr.left);
    } else {
        target = getMinimum(curr.right);

    curr.id = target.id;

    target.parent = null;
};

Tree.prototype.isRoot = function (node) {
    return node.parent === -1;
};

Tree.prototype.preorder = function () {
    const queue = [this.root];
    const preorder = [];
    while (queue.length) {
        const curr = queue.shift();
        preorder.push(curr);
        if (curr.right) queue.unshift(curr.right);
        if (curr.left) queue.unshift(curr.left);
    }
    return preorder;
};

Tree.prototype.inorder = function () {
    const queue = [];
    let curr = this.root;
    const inorder = [];

    while (curr || queue.length) {
        while (curr) {
            queue.push(curr);
            curr = curr.left;
        }
        curr = queue.pop();
        inorder.push(curr);
        curr = curr.right;
    }
    return inorder;
};

function Node(id) {
    if (!new.target) return new Node(id);
    this.id = id;
    this.left = null;
    this.right = null;
    this.parent = null;
}
