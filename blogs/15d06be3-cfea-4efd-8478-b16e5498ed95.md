function swap(list, m, n) {
    const temp = list[m];
    list[m] = list[n];
    list[n] = temp;
}

function bubbleSort(n, list) {
    list = [...list];
    let flag = 1;
    let i = 0;
    while (flag) {
        flag = 0;
        for (let j = n - 1; j > i; j--) {
            if (list[j][0] < list[j - 1][0]) {
                flag = 1;
                swap(list, j, j - 1);
            }
        }
        i++;
    }
    return list;
}

function multiverse(M, N, universes) {
    let answer = 0;
    const map = {};
    for (let i = 0; i < M; i++) {
        const universe = universes[i].map((e, i) => [e, i]);
        universe.sort((a, b) => a[0] - b[0]);
        //const key = bubbleSort(N, universe).reduce((a, c) => a + c[1], '');
        const key = universe.reduce((a, c) => a + c[1], '');
        if (!map[key]) map[key] = 0;
        map[key]++;
        if (map[key] > 1) {
            answer++;
            delete map[key];
        }
    }
    return answer;
}

function multiverse(M, N, universes) {
    let answer = 0;
    const map = {};
    for (let i = 0; i < M; i++) {
        const uni = universes[i];
        for (let j = 0; j < N; j++) {
            uni[j] = [j, uni[j]];
        }
    }
    for (let i = 0; i < M; i++) {
        const key = bubbleSort(universes[i])
            .map((e) => e[0])
            .join('');
        if (!map[key]) map[key] = 0;
        map[key]++;
        if (map[key] >= 2) {
            answer++;
            map[key] = 0;
        }
    }
