let input = [];

        input: process.stdin,

    })
    .on('line', function (line) {
        input.push(line);

    });

function solution(input) {
    const n = Number(input.shift());

    console.log(
        Array(n)
            .fill(0)
            .map((_, idx) => idx + 1)
            .reduce((a, c) => a * c, 1),
    );
    //console.log(Factorial(n));
}

function Factorial(n) {
    if (n === 1) return 1;

    return n * Factorial(n - 1);
}
