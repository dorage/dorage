const makeHeap = (arr, condition) => {
    const parentIdx = (n) => Math.floor((n - 1) / 2);
    
    const childIdx = (n) => [n * 2 + 1, n * 2 + 2];
    
    const swap = (a, b) => {
        [arr[a], arr[b]] = [arr[b], arr[a]];
    };
    
    const heapify = (i = 0) => {
        const [l, r] = childIdx(i);
        let curr = i;
        
        if(l < arr.length && condition(arr[curr], arr[l])) {
            curr = l;
        }
        if(r < arr.length && condition(arr[curr], arr[r])) {
            curr = r;
        }
        
        if(curr !== i) {
            swap(curr, i);
            heapify(curr);
        }
    };
    
    const elevate = () => {
        let curr = arr.length - 1;
        let parent = parentIdx(curr);
        while(curr > 0 && condition(arr[parent], arr[curr])) {
            swap(parent, curr);
            curr = parent;
            parent = parentIdx(parent);
        }
    };
    
    const push = (v) => {
        arr.push(v);
        elevate();
    };
    
    const pop = () => {
        swap(0, arr.length - 1);
        const v = arr.pop();
        heapify();
        return v;
    }
    
    return { push, pop };
};

/**
 * @param {number[]} arr
 * @param {number} k
 * @param {number} x
 * @return {number[]}
 */
var findClosestElements = function(arr, k, x) {
    const answer = [];
    const heap = [];
    const condition = (p, c) => {
        return (Math.abs(c - x) < Math.abs(p - x)) || ((Math.abs(c - x) === Math.abs(p - x)) && c < p);
    };
    const { push, pop } = makeHeap(heap, condition);
    
    for(const n of arr) {
        push(n);
    }
    
    for(let i = 0; i < k; i++) {
        answer.push(pop());
    }
    
    answer.sort((a,b) => a - b);
    
    return answer;
};
