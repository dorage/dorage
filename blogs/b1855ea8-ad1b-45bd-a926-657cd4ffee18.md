/**
 * @param {number[][]} isConnected
 * @return {number}
 */
var findCircleNum = function(isConnected) {
    let province = 0;
    const n = isConnected.length;
    const visited = Array(n).fill(false);
    
    for(let i = 0; i < n; i++) {
        if(visited[i]) continue;
        visited[i] = true;
        
        province++;
        
        const q = [i];
        while(q.length) {
            const preq = [];
            while(q.length) {
                const idx = q.shift();
                
                for(let j = 0; j < n; j++) {
                    if(!isConnected[idx][j] || visited[j]) continue;
                    visited[j] = true;
                    preq.push(j);
                }
            }
            q.push(...preq);
        }
    }
    
    return province;
};
