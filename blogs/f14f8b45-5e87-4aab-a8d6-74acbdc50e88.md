const readline = require('readline');

let input = [];

readline
    .createInterface({
        input: process.stdin,
        output: process.stdout,
    })
    .on('line', function (line) {
        input.push(line);
    })
    .on('close', function () {
        solution(input);
        process.exit();
    });

function solution(input) {
    const n = Number(input.shift());

    const tree = Array(n)
        .fill(null)
        .map((_, idx) => Node(idx));

    // left, right ì‚½ì…
    for (const line of input) {
        const [id, left, right] = line.split(' ').map(Number);

        // left
        if (left !== -1) {
            tree[id].left = tree[left];
            tree[left].parent = tree[id];
        }
        // right
        if (right !== -1) {
            tree[id].right = tree[right];
            tree[right].parent = tree[id];
        }
    }

    console.log('Preorder');
    console.log(
        ' ' +
            preorder(tree[0].getRoot())
                .map((node) => node.id)
                .join(' '),
    );
    console.log('Inorder');

    console.log(
        ' ' +
            inorder(tree[0].getRoot())
                .map((node) => node.id)
                .join(' '),
    );
    console.log('Postorder');

    console.log(
        ' ' +
            postorder(tree[0].getRoot())
                .map((node) => node.id)
                .join(' '),
    );
}

function preorder(node) {
    let L = node.left ? preorder(node.left) : [];
    let R = node.right ? preorder(node.right) : [];

    return [node, ...L, ...R];
}
function inorder(node) {
    let L = node.left ? inorder(node.left) : [];
    let R = node.right ? inorder(node.right) : [];

    return [...L, node, ...R];
}
function postorder(node) {
    let L = node.left ? postorder(node.left) : [];
    let R = node.right ? postorder(node.right) : [];

    return [...L, ...R, node];
}

function Node(id) {
    if (!new.target) return new Node(id);
    this.id = id;
    this.parent = null;
    this.left = null;
    this.right = null;
}
Node.prototype.getRoot = function () {
    if (!this.parent) return this;
    return this.parent.getRoot();
};

function log(id, obj) {
    return (
        `node ${id}: ` +
        Object.keys(obj)
            .map((key) => {
                if (key === 'type') {
                    return `${obj[key]}`;
                }
                return `${key} = ${obj[key]}`;
            })
            .join(', ')
    );
}
