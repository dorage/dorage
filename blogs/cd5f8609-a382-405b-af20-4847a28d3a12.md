/**
 * @param {number[]} nums
 * @param {number} target
 * @return {number}
 */
var combinationSum4 = function(nums, target) {
    const n = nums.length;
    const dp = Array(target + 1).fill(0);
    dp[0] = 1;
    
    for(let i = 1; i <= target; i++) {
        for(let j = 0; j < n; j++) {
            const num = nums[j];
            
            if(i < num) continue;
            
            dp[i] += dp[i - num];
        }
    }
    
    return dp[target];
};
