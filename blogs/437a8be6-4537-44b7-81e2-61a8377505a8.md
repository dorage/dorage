const dec2bin = (dec) => (dec >>> 0).toString(2);
const bitcount = (dec) => [...dec2bin(dec)].reduce((a,c) => c === '1' ? a + 1 : a, 0);

/**
 * @param {number} num1
 * @param {number} num2
 * @return {number}
 */
var minimizeXor = function(num1, num2) {
    let a = bitcount(num1);
    let b = bitcount(num2);
    let res = num1;
    
    for(let i = 0; i < 32; i++) {
        if (a === b) break;
        if (a > b && ((1 << i) & num1) > 0) {
            res ^= 1 << i;
            a--;
        }
        if (a < b && ((1 << i) & num1) === 0) {
            res ^= 1 << i;
            a++;
        }
    }
    
    return res;
};
