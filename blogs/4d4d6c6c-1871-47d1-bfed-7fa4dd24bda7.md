 * @param {number} amount
 * @param {number[]} coins

var change = function(amount, coins) {
    const n = coins.length;
    const dp = Array(n + 1).fill().map(()=>Array(amount + 1).fill(0));
    
    for(let i = 1; i <= n; i++) {
        const coin = coins[i - 1];
        dp[i][0] = 1;
        for(let j = 1; j <= amount; j++) {
            dp[i][j] = dp[i - 1][j] + (j - coin >= 0 ? dp[i][j - coin] : 0);
        }
    }
    
    return dp[n][amount];
};
