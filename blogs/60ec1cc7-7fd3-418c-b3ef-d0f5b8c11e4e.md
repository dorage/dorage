function rob(nums: number[]): number {
    let answer = [0, nums[0]];
    for(let i = 1; i < nums.length; i++) {
        answer = [answer[1], Math.max(answer[0] + nums[i], answer[1])];
    }
    return Math.max(...answer);
};
