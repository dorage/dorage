```tsx
class Solution:
    def champagneTower(self, poured: int, query_row: int, query_glass: int) -> float:
        x = query_glass + 1
        y = query_row + 1

        t = [[poured]]

        for i in range(y):
            for j in range(min(i + 1, x)):
                if len(t) <= i:
                    t.append([0] * i + 1)
                if len(t) <= i + 1:
                    t.append([0] * (i + 2))

                if t[i][j] > 1:
                    half = (t[i][j] - 1) / 2
                    t[i + 1][j] += half
                    t[i + 1][j + 1] += half
        
        return min(t[query_row][query_glass], 1)
```

```jsx
class Solution:
    def champagneTower(self, poured: int, query_row: int, query_glass: int) -> float:
        t = [poured]

        for i in range(query_row):
            newT = [0] * (i + 2)
            for j in range(i + 1):
                if t[j] > 1:
                    half = (t[j] - 1) / 2
                    newT[j] += half
                    newT[j + 1] += half
            t = newT

        return min(t[query_glass], 1)
```
