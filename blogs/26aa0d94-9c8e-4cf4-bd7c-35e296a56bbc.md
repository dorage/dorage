const input = [];

function solution(input) {
    const [n, m] = input.shift().split(' ').map(Number);
    const G = Array(n)
        .fill(0)
        .map(() => []);

    for (let i = 0; i < m; i++) {
        const [t, s] = input.shift().split(' ').map(Number);
        G[t].push(s);
        G[s].push(t);
    }
    const colorMap = checkConnected(n, G);

    const result = [];
    const q = Number(input.shift());
    for (let i = 0; i < q; i++) {
        const [t, s] = input.shift().split(' ').map(Number);
        result.push(colorMap[t] === colorMap[s] ? 'yes' : 'no');
    }
    console.log(result.join('\n'));
}

function checkConnected(n, G) {
    const colorMap = Array(n).fill(0);
    let id = 1;
    for (let i = 0; i < n; i++) {
        if (colorMap[i]) continue;
        colorMap[i] = id;
        bfs(id, G, colorMap, i);
        id++;
    }
    return colorMap;
}
function dfs(id, G, colorMap, i) {
    const stack = [i];

    while (stack.length) {
        const t = stack.pop();

        for (s of G[t]) {
            if (colorMap[s]) continue;
            colorMap[s] = id;
            stack.push(s);
        }
    }
    return colorMap;
}
function bfs(id, G, colorMap, i) {
    let queue = [i];

    while (queue.length) {
        const t = queue.shift();
        for (const s of G[t]) {
            if (colorMap[s]) continue;
            colorMap[s] = id;

            queue.push(s);
        }
    }
}

function solution(input) {
    const [n, m] = input.shift().split(' ').map(Number);
    const G = Array(n)
        .fill(0)
        .map(() => ({
            nodes: [],
        }));

    for (let i = 0; i < m; i++) {
        const [t, s] = input.shift().split(' ').map(Number);
        G[t].nodes.push(s);
        G[s].nodes.push(t);
        if (G[t] !== G[s]) {
            if (G[t].nodes.length >= G[s].nodes.length) {
                G[t].nodes.push(...G[s].nodes);
                G[s].nodes = G[t].nodes;
                G[s] = G[t];
            } else {
                G[s].nodes.push(...G[t].nodes);
                G[t].nodes = G[s].nodes;
                G[t] = G[s];
            }
        }
    }
    const result = [];
    const q = Number(input.shift());
    for (let i = 0; i < q; i++) {
        const [t, s] = input.shift().split(' ').map(Number);
        result.push(G[t].nodes === G[s].nodes ? 'yes' : 'no');
    }
    console.log(result.join('\n'));
}
