const reverseDigits = (n) => {
    const arr = [...n.toString()];
    arr.reverse();
    return Number(arr.join(''));
}

/**
 * @param {number[]} nums
 * @return {number}
 */
var countDistinctIntegers = function(nums) {
    const set = new Set();
    
    for(const n of nums) {
        set.add(n);
        set.add(reverseDigits(n));
    }
    
    return set.size;
};
