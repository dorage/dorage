var StockSpanner = function() {
    this.stack = [[Infinity, 0]];
};

/** 
 * @param {number} price
 * @return {number}
 */
StockSpanner.prototype.next = function(price) {
    let count = 1;
    while(this.stack.at(-1)[0] <= price) {
        count += this.stack.pop()[1];
    }

    this.stack.push([price, count]);
    

    return this.stack.at(-1)[1];
};
