var maxScoreSightseeingPair = function(values) {
    let max = values[0];
    let answer = -Infinity;
    
    for(let i = 1; i < values.length; i++) {
        max = Math.max(max, values[i - 1] + (i - 1));
        answer = Math.max(answer, max + values[i] - i);
    }
    
    return answer;
};
