 * @param {number[]} nums

var threeSum = function (nums) {
    nums.sort((a, b) => a - b);
    const answer = [];

    for (let i = 0; i < nums.length; i++) {
        const head = nums[i];
        let l = i + 1;
        let r = nums.length - 1;

        while (l < r) {
            const sum = nums[l] + nums[r];

            if (head + sum === 0) {
                const triplet = [nums[i], nums[l], nums[r]];
                answer.push(triplet);

                while (l < r && nums[l] === triplet[1]) {
                    l++;
                }
                while (l < r && nums[r] === triplet[2]) {
                    r--;
                }
            } else if (head + sum  < 0) {
                l++;
            } else if (head + sum  > 0) {
                r--;
            }
        }

        while (i + 1 < nums.length && nums[i] === nums[i + 1]) {
            i++;
        }
    }

    return answer;
};
