const insert = (n) => {
    const parentIdx = (idx) => Math.floor((idx - 1) * 0.5);

    heap.push(n);
    const len = heap.length;
    let curr = len - 1;
    let parent = parentIdx(curr);

    while (curr > 0 && compare(curr, parent)) {
        swap(curr, parent);
        curr = parent;
        parent = parentIdx(curr);

const extract = () => {
    swap(0, heap.length - 1);
    const root = heap.pop();

    let i = 0;
    while (true) {
        const left = 2 * i + 1;
        const right = 2 * i + 2;
        let minIdx = i;
        if (left < heap.length && compare(left, minIdx)) {
            minIdx = left;
        }
        if (right < heap.length && compare(right, minIdx)) {
            minIdx = right;
        }
        if (i !== minIdx) {
            swap(i, minIdx);
            i = minIdx;
            continue;
        }
        break;
    }
    return root;
};

function solution(input) {
    const parse10 = (n) => parseInt(n, 10);
    const n = Number(input.shift());
    const G = Array(n);
    for (let i = 0; i < n; i++) {
        const [u, k, ...adjs] = input.shift().split(' ').map(parse10);
        G[u] = [];
        for (let j = 0; j < k; j++) {
            const v = adjs[2 * j];
            const c = adjs[2 * j + 1];
            G[u][v] = c;
        }
    }

    return dijkstra(n, G);
}

function dijkstra(n, G) {
    const d = Array(n).fill(Infinity);
    let count = 0;

    heap.push([0, 0]);

    while (count < n) {
        const node = extract();
        const [u, cost] = node;
        if (d[u] < Infinity) continue;
        d[u] = cost;
        count++;

        G[u].forEach((e, idx) => {
            insert([idx, cost + e]);
        });
    }

        console.log(`${i} ${d[i]}`);
    }
}
