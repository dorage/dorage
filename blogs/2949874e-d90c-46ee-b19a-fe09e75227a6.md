function uniquePaths(m: number, n: number): number {
    const map = Array(m + 1).fill(0).map(() => Array(n + 1).fill(1));
    
    for(let i = 2; i < m + 1; i++){
        for(let j = 2; j < n + 1; j++){
            map[i][j] = map[i - 1][j] + map[i][j - 1];
        }
    }
    
    return map[m][n];
};
