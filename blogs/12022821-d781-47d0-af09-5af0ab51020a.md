/**
 * @param {number[]} arr
 * @return {boolean}
 */
var threeConsecutiveOdds = function(arr) {
    let odd = 0;
    for(let i = 0, len = arr.length; i < len; i++){
        if(arr[i] % 2) {
            odd++;
            if(odd >= 3) return true;
            continue;
        }
        odd = 0;
    }
    return false;
};
