/**
 * @param {string[]} words
 * @param {number} k
 * @return {string[]}
 */
var topKFrequent = function(words, k) {
    const map = new Map();
    for(const word of words) {
        if(!map.has(word)) map.set(word, 0);
        map.set(word, map.get(word) + 1);
    }
    
    const arr = [...map];
    arr.sort((a, b) => b[1] === a[1] ? (a[0] > b[0] ? 1 : -1) : b[1] - a[1]);
    return arr.slice(0, k).map((e) => e[0]);
};
