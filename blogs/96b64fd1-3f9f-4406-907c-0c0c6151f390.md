function solution(input) {
    const setInfinity = (n) => {
        return n === -1 ? 2001 : n;
    };

    const n = Number(input.shift());
    const G = input.map((l) => l.split(' ').map(Number));
    G.forEach((l) => l.shift());
    return prim(n, G).reduce((a, c) => a + c);
}
