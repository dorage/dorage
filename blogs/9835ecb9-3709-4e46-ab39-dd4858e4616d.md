const getMN = (mat) => [mat.length, mat[0].length];

/**
 * @param {character[][]} matrix
 * @return {number}
 */
var maximalSquare = function(matrix) {
    const [m, n] = getMN(matrix);
    let answer = 0;
    
    for(let i = 0; i < m; i++) {
        for(let j = 0; j < n; j++) {
            if(matrix[i][j] === "0") continue;
            if(!i || !j) {
                answer = Math.max(answer, 1);
                continue;
            }
            
            matrix[i][j] = Math.min(
                1 * matrix[i - 1][j],
                1 * matrix[i][j - 1],
                1 * matrix[i - 1][j - 1],
            ) + 1;
            
            answer = Math.max(answer, matrix[i][j] ** 2);
        }
    }
    
    return answer;
};
