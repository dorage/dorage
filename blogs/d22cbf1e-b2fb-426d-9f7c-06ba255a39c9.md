# """
# This is MountainArray's API interface.
# You should not implement it, or speculate about its implementation
# """
#class MountainArray:
#    def get(self, index: int) -> int:
#    def length(self) -> int:

class Solution:
    def findInMountainArray(self, target: int, mountain_arr: 'MountainArray') -> int:
        n = mountain_arr.length()
        
        l, r = 0, n -1
        memo = [-1] * n
        peak = -1

        while l < r:
            mid = floor((l + r)/2)
            if mid - 1 >= 0 and memo[mid - 1] == -1: memo[mid - 1] = mountain_arr.get(mid - 1)
            if mid + 1 < n and memo[mid + 1] == -1: memo[mid + 1] = mountain_arr.get(mid + 1)
            if memo[mid] == -1: memo[mid] = mountain_arr.get(mid)
            if memo[mid - 1] < memo[mid] and memo[mid + 1] < memo[mid]:
                peak = mid
                break
            elif memo[mid - 1] < memo[mid]:
                l = mid + 1
            elif memo[mid] > memo[mid + 1]:
                r = mid

        l, r = 0, peak + 1
        while l < r:
            mid = floor((l + r)/2)
            
            if memo[mid] == -1: memo[mid] = mountain_arr.get(mid)
            
            if memo[mid] > target:
                r = mid
            elif memo[mid] < target:
                l = mid + 1
            else:
                return mid
        
        l, r = peak, n
        while l < r:
            mid = floor((l + r)/2)
            
            if memo[mid] == -1: memo[mid] = mountain_arr.get(mid)
            
            if memo[mid] > target:
                l = mid + 1
            elif memo[mid] < target:
                r = mid
            else:
                return mid

        return -1
