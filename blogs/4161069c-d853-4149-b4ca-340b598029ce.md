var minCost = function(colors, neededTime) {
    let answer = 0;
    let r = 0; 
    let l = 0;
    
    while(r < colors.length) {
        while(colors[l] === colors[r]) {
            r++;
        }
        
        if(r - l > 1) answer += minTime(neededTime.slice(l, r));
        
        l = r;
    }
    
    return answer;
};

const isOdd = (n) => n % 2 === 1;

const minTime = (times) => {
    let max = 0;
    let sum = 0;
    
    for(const n of times) {
        max = Math.max(n, max);
        sum += n;
    }
    
    return sum - max;
};
