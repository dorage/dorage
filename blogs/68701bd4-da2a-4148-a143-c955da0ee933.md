const divByHalfN = (n) => (e) => e / 2 ** (n - 1);

function Z(n, r, c) {
    if (n <= 0) return 0;
    const [remainR, remainC] = [r, c].map(divByHalfN(n)).map(Math.floor);
    const x = 2 ** (2 * n) / 4;
    const y = 2 ** (n - 1);
    // 1
    if (remainR === 0 && remainC === 0) {
        return 0 + Z(n - 1, r, c);
    }
    // 2
    if (remainR === 0 && remainC !== 0) {
        return 1 * x + Z(n - 1, r, c - y);
    }
    // 3
    if (remainR !== 0 && remainC === 0) {
        return 2 * x + Z(n - 1, r - y, c);
    }
    // 4
    if (remainR !== 0 && remainC !== 0) {
        return 3 * x + Z(n - 1, r - y, c - y);
    }
}
