/**
 * @param {number[][]} boxTypes
 * @param {number} truckSize
 * @return {number}
 */
var maximumUnits = function(boxTypes, truckSize) {
    let answer = 0;
    boxTypes.sort((a,b)=>b[1]-a[1]);
    for(let [num, unit] of boxTypes){
        if(truckSize < num) num = truckSize;
        answer += unit * num;
        truckSize -= num;
        if(!truckSize) break;
    }
    return answer;
};
