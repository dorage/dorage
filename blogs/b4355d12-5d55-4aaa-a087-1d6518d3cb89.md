const reverseDigits = (n) => {
    const arr = [...n.toString()];
    arr.reverse();
    return Number(arr.join(''));
}

/**
 * @param {number} num
 * @return {boolean}
 */
var sumOfNumberAndReverse = function(num) {
    
    for(let i = 0; i <= num; i++) {
        if(i + reverseDigits(i) === num) return true;
    }
    
    return false;
};
