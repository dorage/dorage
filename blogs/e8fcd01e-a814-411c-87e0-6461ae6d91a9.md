```jsx
const readline = require('readline');

let input = [];

readline
    .createInterface({
        input: process.stdin,
        output: process.stdout,
    })
    .on('line', function (line) {
        input.push(line);
    })
    .on('close', function () {
        solution(input);
        process.exit();
    });

function solution(input) {
    const n = Number(input.shift());
    const inorder = input.shift().split(' ').map(Number);
    const postorder = input.shift().split(' ').map(Number);
    const queue = [[0, n, 0, n]];
    const result = [];

    while (queue.length) {
        const [startIn, endIn, startPost, endPost] = queue.pop();
        if (endIn - startIn < 1) continue;

        const root = postorder[endPost - 1];
        let rootIdxInorder;
        for (let i = startIn; i < endIn; i++) {
            if (inorder[i] === root) {
                rootIdxInorder = i;
                break;
            }
        }
        lengthOfLeftInorder = rootIdxInorder - startIn;

        result.push(root);
        // RIGHT
        queue.push([
            rootIdxInorder + 1,
            endIn,
            startPost + lengthOfLeftInorder,
            endPost - 1,
        ]);
        // LEFT
        queue.push([
            startIn,
            rootIdxInorder,
            startPost,
            startPost + lengthOfLeftInorder,
        ]);
    }
    console.log(result.join(' '));
}
```
