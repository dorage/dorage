function DFS(G, DF, visited, i, time) {
    if (visited.includes(i)) return time;
    visited.push(i);

    const prevTime = ++time;

    for (const j of G[i]) {
        time = DFS(G, DF, visited, j, time);

    const currTime = ++time;
    DF[i] = [prevTime, currTime];

    return currTime;
}

function deepCopy(l) {
    return l.map((e) => (e instanceof Array ? deepCopy(e) : e));
}

// loop
function DFS_loop(n, G) {
    G = deepCopy(G);
    const D = Array(n + 1).fill(0);
    const F = Array(n + 1).fill(0);
    const visited = Array(n + 1).fill(false);
    let time = 0;

    for (let i = 1; i <= n; i++) {
        if (visited[i]) continue;
        visited[i] = true;

        const stack = [i];
        D[i] = ++time;

        while (stack.length) {
            const u = stack[stack.length - 1];

            if (G[u].length) {
                const v = G[u].shift();
                if (visited[v]) continue;
                visited[v] = true;

                stack.push(v);
                D[v] = ++time;
            } else {
                stack.pop();
                F[u] = ++time;
            }
        }
    }
    return [D, F];
}

function DFS_recursion(n, G) {
    G = deepCopy(G);
    const D = Array(n + 1).fill(0);
    const F = Array(n + 1).fill(0);
    const visited = Array(n + 1).fill(false);

    let time = 0;
    for (let i = 1; i <= n; i++) {
        time = recursion(G, D, F, visited, i, time);
    }
    return [D, F];
}

// recursion
function recursion(G, D, F, visited, i, time) {
    if (visited[i]) return time;
    visited[i] = true;

    D[i] = ++time;

    for (const j of G[i]) {
        time = recursion(G, D, F, visited, j, time);
    }

    F[i] = ++time;

    return time;
}
