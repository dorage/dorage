    .createInterface({
        input: process.stdin,
        output: process.stdout,

    .on('line', function (line) {
        input.push(line);
    })
    .on('close', function () {
        solution(input);
        process.exit();
    });

function solution(input) {
    const n = Number(input.shift());
    const tree = [];
    for (let i = 0; i < n + 1; i++) {
        tree.push(Node(i));
        if (i === 1) tree[i].parent = true;
    }
    const lines = input.map((e) => e.split(' ').map(Number));
    while (lines.length) {
        const [id1, id2] = lines.shift();
        const node1 = tree[id1];
        const node2 = tree[id2];

        if (!node1.parent && !node2.parent) {
            lines.push([id1, id2]);
            continue;
        }

        if (!node1.parent) {
            node1.parent = node2;
            continue;
        }
        node2.parent = node1;
    }

    console.log(
        tree
            .slice(2, n + 1)
            .map((e) => e.parent.id)
            .join('\n'),
    );
}

function Node(id) {
    if (!new.target) return new Node(id);
    this.id = id;
    this.parent = null;
}

        input.push(line);

        process.exit();

function solution(input) {
    const n = Number(input.shift());
    const parents = [];
    const graph = [];
    // root node
    parents[1] = true;

    for (const line of input) {
        const [n1, n2] = line.split(' ').map(Number);

        if (!graph[n1]) graph[n1] = [];
        if (!graph[n2]) graph[n2] = [];
        graph[n1].push(n2);
        graph[n2].push(n1);
    }

    const queue = [1];
    const visited = [];

    while (queue.length) {
        const id = queue.shift();
        visited.push(id);

        for (const adjacentId of graph[id]) {
            if (!parents[adjacentId]) {
                parents[adjacentId] = id;
            }
        }
        queue.push(...graph[id].filter((e) => !visited.includes(e)));
    }

    console.log(parents.slice(2, n + 1).join('\n'));
}
