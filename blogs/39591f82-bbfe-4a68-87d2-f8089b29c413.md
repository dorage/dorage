/**
 * @param {number[]} nums
 * @return {number}
 */
var minimizeArrayValue = function(nums) {
    const len = nums.length;
    let norm = nums[0];
    
    let space = 0;
    
    for(let i = 1; i < len; i++) {
        const n = nums[i]
        const diff = n - norm;
        
        if(n <= norm) {
            space += Math.abs(diff);
            continue;
        }
        
        if(diff <= space) {
            space -= diff;
            continue;
        }
        
        const remainder = Math.floor((diff - space) / (i + 1));
        const quotient = Math.floor((diff - space) % (i + 1));
        
        space = quotient ? (i + 1) - quotient : 0;
        norm += remainder + (quotient ? 1 : 0);
    }
    
    return norm;
};
