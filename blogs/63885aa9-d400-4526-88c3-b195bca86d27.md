/**
 * @param {number[]} asteroids

var asteroidCollision = function(asteroids) {
    const stack = [];
    
    const condition = (a) => stack.length && a < 0 && stack[stack.length - 1] > 0;
    
    forloop:
    for(const a of asteroids) {
        
        if(condition(a)) {
            while(condition(a)) {
                const absA = Math.abs(a);
                const b = stack.pop();
                if(absA > b) continue;
                if(absA < b) stack.push(b);
                continue forloop;
            }
        }
        stack.push(a);
    }
    
    return stack;
};
