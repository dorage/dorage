const clamp = (n:number, min:number, max:number) => Math.min(Math.max(min, n), max);

function myAtoi(s: string): number {
    let curr = [];
    let sign = 1;
    let typed = false;
    for (let i = 0; i < s.length; i++) {
        const c = s[i];
        if (c === ' ') {
            if(typed) break;
            continue;
        }
        if (c === '+' || c === '-') {
            if(typed) break;
            sign = c === '+' ? 1 : -1;
            typed = true;
            continue;
        }
        if(isNaN(Number(c))){
            break;
        }
        curr.push(c);
        typed = true;
    }
    const min = -1 * 2 ** 31;
    const max = 2 ** 31 - 1;
    
    return clamp(Number(curr.join('')) * sign, min, max);
}
