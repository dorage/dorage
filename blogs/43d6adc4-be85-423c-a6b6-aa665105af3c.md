const getMN = (mat) => {
    return [mat.length, mat[0].length]
}

const getDiagonalArray = (mat, x, y) => {
    const [m, n] = getMN(mat);
    const vals = [];
    while(x < n && y < m){
        vals.push(mat[y][x]);
        x++, y++;
    }
    return vals;
}
const setDiagonalArray = (mat, x, y, vals) => {
    const [m, n] = getMN(mat);
    while(x < n && y < m){
        mat[y][x] = vals.shift();
        x++, y++;
    }
}
/**
 * @param {number[][]} mat
 * @return {number[][]}
 */
var diagonalSort = function(mat) {
    const [m, n] = getMN(mat);
    for(let i = 1; i < m + n; i++) {
        const y = Math.max(0, m - i);
        const x = Math.max(0, i - m);
        const arr = getDiagonalArray(mat,x,y);
        arr.sort((a,b)=>a-b);
        setDiagonalArray(mat,x,y,arr);
    }
    return mat;
};
