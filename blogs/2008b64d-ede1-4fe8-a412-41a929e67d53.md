        input.push(line);

        process.exit();
    });

function solution(input) {
    const n = Number(input.shift());

    console.log(Fibonacci(n));
}

function Fibonacci(n) {
    if (n === 0) return 0;
    if (n === 1) return 1;
    return Fibonacci(n - 1) + Fibonacci(n - 2);
}
