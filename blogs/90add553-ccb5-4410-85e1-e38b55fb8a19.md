function fill(n, k, capacity, weights) {
    let i = 0;
    for (let j = 0; j < k; j++) {
        let s = 0;
        if ((s, weights[i], capacity))
            while (s + weights[i] <= capacity) {
                s += weights[i];
                i++;
                if (i === n) return n;
            }
    }
    return i;
}

function Allocation(n, k, weights) {
    let left = 0;
    let right = 100000 * 10000;

    while (right - left > 1) {
        const mid = Math.floor((left + right) / 2);

        const v = fill(n, k, mid, weights);

        if (v >= n) right = mid;
        else left = mid;
    }

    return right;
}
