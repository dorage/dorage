```tsx
function numberOfWeakCharacters(properties: number[][]): number {
    properties.sort((a,b) => a[0] === b[0] ? b[1] - a[1] : a[0] - b[0]);
    
    let answer = 0;
    let mtn = 0;
    
    for(let i = properties.length - 1; i >= 0; i--)
    {
        const [_, def] = properties[i];
        if(def < mtn) {
            answer++;
        }
        mtn = Math.max(mtn, def);
    }
    
    return answer;
};
```
